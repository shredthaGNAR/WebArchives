Github search on Google

Adds a button to search github.com with Google

Short link: https://greasyfork.org/scripts/462358(Copy)

Reinstall version 2.1.0?
Ask a question, post a review, or report the script. 
  1. // ==UserScript==
  2. // @name         Github search on Google
  3. // @version      2.1.0
  4. // @description  Adds a button to search github.com with Google
  5. // @author       Alexyoe
  6. // @namespace    https://github.com/Alexyoe/Github-on-Google-Search
  7. // @license      MIT
  8. // @include      http*://www.google.*/search*
  9. // @include      http*://google.*/search*
  10. // @run-at       document-end
  11. // ==/UserScript==
  12.  
  13. // Settings
  14. const settings = {
  15.   // Choose exactly one: "icon" or "label"
  16.   displayMode: "icon",
  17.   btnPosition: "start", // "start", "end", or "afterai"
  18.   fixSize: false,
  19. };
  20.  
  21. // Start Code
  22. const queryRegex = /q=[^&]+/g;
  23. const siteRegex = /\+site(?:%3A|\:).+\.[^&+]+/g;
  24. const githubUrl = "+site%3Agithub.com";
  25.  
  26. // Github SVG
  27. let githubIcon =
  28.   '<svg class="DCxYpf" fill="currentColor" xmlns="http://www.w3.org/2000/svg" width="16" height="16" viewBox="0 0 16 16"><path d="M8 0C3.58 0 0 3.58 0 8c0 3.54 2.29 6.53 5.47 7.59.4.07.55-.17.55-.38 0-.19-.01-.82-.01-1.49-2.01.37-2.53-.49-2.69-.94-.09-.23-.48-.94-.82-1.13-.28-.15-.68-.52-.01-.53.63-.01 1.08.58 1.23.82.72 1.21 1.87.87 2.33.66.07-.52.28-.87.51-1.07-1.78-.2-3.64-.89-3.64-3.95 0-.87.31-1.59.82-2.15-.08-.2-.36-1.02.08-2.12 0 0 .67-.21 2.2.82.64-.18 1.32-.27 2-.27.68 0 1.36.09 2 .27 1.53-1.04 2.2-.82 2.2-.82.44 1.1.16 1.92.08 2.12.51.56.82 1.27.82 2.15 0 3.07-1.87 3.75-3.65 3.95.29.25.54.73.54 1.48 0 1.07-.01 1.93-.01 2.2 0 .21.15.46.55.38A8.012 8.012 0 0 0 16 8c0-4.42-3.58-8-8-8z"/></svg>';
  29. const isImageSearch = /[?&]tbm=isch/.test(location.search);
  30.  
  31. // Trusted Types for CSP
  32. if (typeof trustedTypes !== "undefined") {
  33.   const p = trustedTypes.createPolicy("html", { createHTML: (x) => x });
  34.   githubIcon = p.createHTML(githubIcon);
  35. }
  36.  
  37.  
  38. // Main function to run on load. Waits for nav to load.
  39. (function waitForNav() {
  40.   const nav = Array.from(
  41.     document.querySelectorAll('div[role="navigation"]')
  42.   ).find((n) => n.querySelector('div[role="listitem"] a'));
  43.   if (!nav) return setTimeout(waitForNav, 200);
  44.  
  45.   // Grab the first wrapper <div role="listitem"> that isn't selected or "AI Mode"
  46.   const sampleItem = Array.from(
  47.     document.querySelectorAll('div[role="listitem"]')
  48.   ).find((item) => {
  49.     const isSelected = item.querySelector('[selected], [aria-current="page"]');
  50.     const text = item.textContent.trim();
  51.     return !isSelected && text !== "AI Mode";
  52.   });
  53.   if (!sampleItem) return; // bail if nothing there
  54.  
  55.   // Clone the entire wrapper.
  56.   const newItem = sampleItem.cloneNode(true);
  57.  
  58.   // Inside that clone, find the <a>
  59.   const btn = newItem.querySelector("a");
  60.   btn.href = window.location.href.replace(queryRegex, (m) =>
  61.     m.search(siteRegex) >= 0 ? m.replace(siteRegex, githubUrl) : m + githubUrl
  62.   );
  63.  
  64.   // find their inner div (jsname) or fallback to the <a>
  65.   const inner = btn.querySelector("div[jsname]") || btn;
  66.  
  67.   // Clear the inner div
  68.   inner.innerHTML = "";
  69.  
  70.   if (settings.displayMode === "label") {
  71.     const textWrapper = document.createElement("span");
  72.     textWrapper.textContent = "Github";
  73.     textWrapper.className = "R1QWuf";
  74.     inner.appendChild(textWrapper);
  75.   } else {
  76.     const iconWrapper = document.createElement("span");
  77.     iconWrapper.className = "R1QWuf";
  78.     iconWrapper.style.lineHeight = "17px";
  79.     iconWrapper.innerHTML = githubIcon;
  80.     inner.appendChild(iconWrapper);
  81.   }
  82.  
  83.   // Insert the wrapper at the requested position
  84.   const first = nav.querySelector(':scope div[role="listitem"]');
  85.   if (settings.btnPosition === "start") {
  86.     first ? first.before(newItem) : nav.prepend(newItem);
  87.   } else if (settings.btnPosition === "end") {
  88.     nav.append(newItem);
  89.   } else {
  90.     first ? first.after(newItem) : nav.append(newItem);
  91.   }
  92.  
  93.   // optional: prevent wrapping
  94.   if (settings.fixSize) {
  95.     nav.style.maxWidth = "inherit";
  96.     nav.style.overflowX = "auto";
  97.     nav.style.whiteSpace = "nowrap";
  98.   }
  99. })();
  100.
