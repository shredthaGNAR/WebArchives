elements.set(placeHolderElement, element); element.parentElement.replaceChild(placeHolderElement, element); }); }); addEventListener("single-file-on-after-capture-request", () => { Array.from(elements).forEach(([placeHolderElement, element]) => { placeHolderElement.parentElement.replaceChild(element, placeHolderElement); }); elements.clear(); }); })(); This script converts the title to kebab-case and restores it after the page is processed. // ==UserScript== // @name Convert title to kebab-case // @namespace https://github.com/gildas-lormeau/SingleFile // @version 1.0 // @description [SingleFile] Convert title to kebab-case // @author Gildas Lormeau // @match *://*/* // @noframes // @grant none // ==/UserScript== (() => { let title;
